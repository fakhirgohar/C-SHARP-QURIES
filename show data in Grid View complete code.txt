private void btnSHOW_Click_1(object sender, EventArgs e)
        {
            btnCANCEL.Focus();
            
            string query1;
            string connectionstring = "Data Source=DESKTOP-O0ATRMQ;Initial Catalog=SchoolRecord;Integrated Security=True";
            SqlConnection con = new SqlConnection(connectionstring);

            con.Open();
            string Search = txtSEARCH.Text;

            if (string.IsNullOrEmpty(txtSEARCH.Text.Trim()))
            {

                query1 = "select * from std_rec";

            }
            else
            {
                query1 = $"select * from std_rec where Student_name like '{Search}%'";

            }
            SqlCommand cmd = new SqlCommand(query1, con);
            SqlDataReader reader = cmd.ExecuteReader();

           

            while (reader.Read())
            {
                dataGridView1.Rows.Add(reader["Student_ID"], reader["Student_Name"]);
            }


        }
